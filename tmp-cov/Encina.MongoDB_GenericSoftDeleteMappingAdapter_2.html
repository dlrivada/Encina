 484                // For IVersioned (getter-only), we can still check but not increment
 485                originalVersion = versioned.Version;
 486                filter = BuildVersionedFilter(idFilter, originalVersion.Value);
 487            }
 488            else
 489            {
 490                filter = idFilter;
 491            }
 492
 493            var result = await _collection.ReplaceOneAsync(
 494                filter,
 495                entity,
 496                new ReplaceOptions { IsUpsert = false },
 497                cancellationToken)
 498                .ConfigureAwait(false);
 499
 500            if (result.MatchedCount == 0)
 501            {
 502                // Determine if it's a "not found" or "concurrency conflict"
 503                if (originalVersion.HasValue)
 504                {
 505                    // Check if entity exists with a different version
 506                    var exists = await _collection.CountDocumentsAsync(
 507                        idFilter,
 508                        new CountOptions { Limit = 1 },
 509                        cancellationToken).ConfigureAwait(false);
 510
 511                    if (exists > 0)
 512                    {
 513                        // Entity exists but version doesn't match - concurrency conflict
 514                        var databaseEntity = await _collection.Find(idFilter)
 515                            .FirstOrDefaultAsync(cancellationToken)
 516                            .ConfigureAwait(false);
 517
 518                        var conflictInfo = new ConcurrencyConflictInfo<TEntity>(
 519                            CurrentEntity: entity,
 520                            ProposedEntity: entity,
 521                            DatabaseEntity: databaseEntity);
 522
 523                        return Left<EncinaError, TEntity>(
 524                            RepositoryErrors.ConcurrencyConflict(conflictInfo));
 525                    }
 526                }
 527
 528                return Left<EncinaError, TEntity>(RepositoryErrors.NotFound<TEntity, TId>(id));
 529            }
 530
 531            return Right<EncinaError, TEntity>(entity);
 532        }
 533        catch (MongoException ex)
 534        {
 535            return Left<EncinaError, TEntity>(MapMongoException(ex, "Update"));
 536        }
 537    }

 638    /// <inheritdoc/>
 639    /// <remarks>
 640    /// <para>
 641    /// For entities implementing <see cref="IVersionedEntity"/>, this method performs
 642    /// optimistic concurrency control by including the version in each filter predicate.
 643    /// If any version doesn't match, a concurrency conflict error is returned.
 644    /// </para>
 645    /// <para>
 646    /// The version is automatically incremented for each entity before the update operation.
 647    /// </para>
 648    /// </remarks>
 649    public async Task<Either<EncinaError, Unit>> UpdateRangeAsync(
 650        IEnumerable<TEntity> entities,
 651        CancellationToken cancellationToken = default)
 652    {
 653        ArgumentNullException.ThrowIfNull(entities);
 654
 655        var entityList = entities.ToList();
 656        var originalVersions = new Dictionary<TEntity, long>();
 657
 658        try
 659        {
 660            // Populate audit fields and handle versioning before persistence
 661            foreach (var entity in entityList)
 662            {
 663                AuditFieldPopulator.PopulateForUpdate(entity, _requestContext?.UserId, _timeProvider);
 664
 665                // Store original version and increment for IVersionedEntity
 666                if (entity is IVersionedEntity versionedEntity)
 667                {
 668                    originalVersions[entity] = versionedEntity.Version;
 669                    versionedEntity.Version = versionedEntity.Version + 1;
 670                }
 671                else if (entity is IVersioned versioned)
 672                {
 673                    originalVersions[entity] = versioned.Version;
 674                }
 675            }
 676
 677            var bulkOps = entityList.Select(entity =>
 678            {
 679                var id = _mapping.GetId(entity);
 680                var idFilter = BuildIdFilter(id);
 681
 682                // Use versioned filter if we have original version info
 683                var filter = originalVersions.TryGetValue(entity, out var originalVersion)
 684                    ? BuildVersionedFilter(idFilter, originalVersion)
 685                    : idFilter;
 686
 687                return new ReplaceOneModel<TEntity>(filter, entity) { IsUpsert = false };
 688            }).ToList();
 689
 690            if (bulkOps.Count > 0)
 691            {
 692                var result = await _collection.BulkWriteAsync(bulkOps, cancellationToken: cancellationToken).ConfigureAw
 693
 694                // Check for concurrency conflicts (matched count less than entity count)
 695                if (result.MatchedCount < entityList.Count && originalVersions.Count > 0)
 696                {
 697                    // Find the first entity that wasn't matched (potential concurrency conflict)
 698                    foreach (var entity in entityList)
 699                    {
 700                        var id = _mapping.GetId(entity);
 701                        var idFilter = BuildIdFilter(id);
 702
 703                        var exists = await _collection.CountDocumentsAsync(
 704                            idFilter,
 705                            new CountOptions { Limit = 1 },
 706                            cancellationToken).ConfigureAwait(false);
 707
 708                        if (exists > 0 && originalVersions.ContainsKey(entity))
 709                        {
 710                            // Entity exists but version didn't match - concurrency conflict
 711                            var databaseEntity = await _collection.Find(idFilter)
 712                                .FirstOrDefaultAsync(cancellationToken)
 713                                .ConfigureAwait(false);
 714
 715                            var conflictInfo = new ConcurrencyConflictInfo<TEntity>(
 716                                CurrentEntity: entity,
 717                                ProposedEntity: entity,
 718                                DatabaseEntity: databaseEntity);
 719
 720                            return Left<EncinaError, Unit>(
 721                                RepositoryErrors.ConcurrencyConflict(conflictInfo));
 722                        }
 723                    }
 724                }
 725            }
 726
 727            return Right<EncinaError, Unit>(Unit.Default);
 728        }
 729        catch (MongoException ex)
 730        {
 731            return Left<EncinaError, Unit>(MapMongoException(ex, "UpdateRange"));
 732        }
 733    }

 943    {
 944        var idSelector = _mapping.IdSelector;
 945        var parameter = idSelector.Parameters[0];
 946        var body = Expression.Equal(idSelector.Body, Expression.Constant(id, typeof(TId)));
 947        var lambda = Expression.Lambda<Func<TEntity, bool>>(body, parameter);
 948
 949        return Builders<TEntity>.Filter.Where(lambda);
 950    }

 953    {

 965    /// <summary>
 966    /// Builds a filter that combines the ID filter with a version equality check.
 967    /// </summary>
 968    /// <param name="idFilter">The base ID filter.</param>
 969    /// <param name="expectedVersion">The expected version for optimistic concurrency.</param>
 970    /// <returns>A combined filter that matches both ID and version.</returns>
 971    private static FilterDefinition<TEntity> BuildVersionedFilter(
 972        FilterDefinition<TEntity> idFilter,
